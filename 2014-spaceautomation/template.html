Scripting

rule "inbound Call"
  when Item Incoming_Call_No received update
then 
  val CallType call = Incoming_Call_No.state as CallType
  val String mailSubject =
  "Anruf von Nummer " + call.origNum + " -> " + call.destNum
  sendMail( "us...@domain.de" , mailSubject , "");
end
